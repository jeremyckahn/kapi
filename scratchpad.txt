// API for per-property tweening

var myCircle = kapi.add(circle, {
	x: 10,
	y: 10
});

// Tween myCircle's x property (declared above) to `50` over 2 seconds
myCircle.tween.x.to(50, '2s');

// or...

myCircle.tween.x.to({
	'val': 50,
	'duration': 2000,
	'complete': function() {
		console.log("yay it's done");
	}
});


// another possibility... tweening multiple properties
myCircle.to({
	'x': 50,
	'y': 75,
	'duration': 2000,
	'complete': function() {
		console.log("yay it's done");
	}
});
